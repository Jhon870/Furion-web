/* --- RETROCRAFT AUDIO ENGINE --- */ let audioCtx = null; function initAudio() { if (audioCtx) return; audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } // 1. CLICK DE BOTÓN: Onda cuadrada, caída ultra rápida de tono. function playClick() { initAudio(); const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); osc.type = 'square'; osc.frequency.setValueAtTime(120, audioCtx.currentTime); osc.frequency.exponentialRampToValueAtTime(40, audioCtx.currentTime + 0.05); gain.gain.setValueAtTime(0.1, audioCtx.currentTime); gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.05); osc.connect(gain); gain.connect(audioCtx.destination); osc.start(); osc.stop(audioCtx.currentTime + 0.05); } // 2. TYPING: Blip pixelado corto con frecuencia aleatoria. function playType() { initAudio(); const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); const freq = Math.random() > 0.5 ? 880 : 920; // Alternancia de tono osc.type = 'square'; osc.frequency.setValueAtTime(freq, audioCtx.currentTime); gain.gain.setValueAtTime(0.05, audioCtx.currentTime); gain.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.02); osc.connect(gain); gain.connect(audioCtx.destination); osc.start(); osc.stop(audioCtx.currentTime + 0.02); } // 3. ALARMA/HURT: Sawtooth que cae agresivamente (Sonido "OOF"). function playHurt() { initAudio(); const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); osc.type = 'sawtooth'; osc.frequency.setValueAtTime(160, audioCtx.currentTime); osc.frequency.exponentialRampToValueAtTime(60, audioCtx.currentTime + 0.2); gain.gain.setValueAtTime(0.15, audioCtx.currentTime); gain.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.2); osc.connect(gain); gain.connect(audioCtx.destination); osc.start(); osc.stop(audioCtx.currentTime + 0.2); } // 4. PULSO DE CARGA: Tono puro y brillante (Level Up). function playPulse() { initAudio(); const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); osc.type = 'sine'; osc.frequency.setValueAtTime(1050, audioCtx.currentTime); osc.frequency.exponentialRampToValueAtTime(1200, audioCtx.currentTime + 0.1); gain.gain.setValueAtTime(0.08, audioCtx.currentTime); gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.1); osc.connect(gain); gain.connect(audioCtx.destination); osc.start(); osc.stop(audioCtx.currentTime + 0.1); } // 5. GLITCH/EXPLOSIÓN: Generación de ruido blanco pixelado (Bit-crushed noise). function playGlitch() { initAudio(); const bufferSize = audioCtx.sampleRate * 0.3; // 0.3 segundos const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate); const data = buffer.getChannelData(0); // Generar ruido "cuadrado" aleatorio para efecto Bit-crush let lastVal = 0; for (let i = 0; i < bufferSize; i++) { if (i % 8 === 0) lastVal = Math.random() * 2 - 1; // Sample & Hold data[i] = lastVal; } const noise = audioCtx.createBufferSource(); noise.buffer = buffer; const gain = audioCtx.createGain(); gain.gain.setValueAtTime(0.2, audioCtx.currentTime); gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.3); noise.connect(gain); gain.connect(audioCtx.destination); noise.start(); } /* --- INTEGRACIÓN CON TU INTERFAZ --- */ // Vincular a todos los botones y cartas document.querySelectorAll('.btn-cine, .card').forEach(el => { el.addEventListener('mousedown', playClick); }); // Vincular a los inputs para sonido de escritura document.querySelectorAll('input, textarea').forEach(el => { el.addEventListener('input', playType); }); // Modificar tu función start8sCrash para usar los sonidos const originalCrash = start8sCrash; start8sCrash = async function() { playHurt(); const glitchInterval = setInterval(playGlitch, 400); setTimeout(() => clearInterval(glitchInterval), 8000); return originalCrash.apply(this, arguments); }; // Modificar runProc para el sonido de carga const originalRunProc = runProc; runProc = async function(type) { const pulseInterval = setInterval(playPulse, 200); const result = await originalRunProc.apply(this, arguments); clearInterval(pulseInterval); return result; };
